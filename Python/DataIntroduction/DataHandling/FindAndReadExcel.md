# 원하는 범위 데이터 조회하기


```python
import pandas as pd

df = pd.read_excel('data/~~~.xls', index_col='상품명')		# 엑셀 데이터를 불러오고 인덱스명은 상품명으로 지정

df.head?		# 뒤에 물음표를 붙이면 함수에 대한 설명 나옴, 실제 파이썬에서 적용되지는 않음, jupyter notebook 에서만 적용
df.head()		# 위에서부터 5개 항목 출력
df.tail()		# 아래에서부터 5개 항목 출력
df.nlargest(10, '판매건수')		# 판매건수 기준으로 상위 10개 데이터 출력
df.nsmallest(10, '판매건수')		# 판매건수 기준으로 하위 10개 데이터 출력
df.nsmallest(10, ['판매건수', '가격'])		# 리스트 형식으로 여러 조건도 지정 가능
df.sort_values('가격')		# 가격을 기준으로 정렬해서 출력(default값은 오름차순)
df.sort_values('가격', ascending = False)		# 가격을 기준으로 내림차순 정렬 출력
df.sort_values('가격', ascending = False)[100:200]		# 뒤에 조건을 붙여서 출력 가능, 100번째부터 200번째까지 출력
df.columns		# 컬럼명들 확인

names = ['판매건수', '가격']
df[names]		# 이런식으로 변수에 할당한 뒤 출력하면 데이터프레임 형식으로 출력 가능
df[['판매건수, 가격']]		# 응용하여 이중대괄호로 즉시 표현도 가능
df.loc['야채']		# index를 통한 행단위 검색
df.loc['야채', '딸기','봉투']		# 다수의 조건을 통한 검색도 가능하고 위처럼 변수에 담아서도 가능

names = ['야채', '딸기', '봉투']
df.loc[names].sort_values('판매건수', ascending = False)		# 함수 조합 응용

df.iloc[:10]		# 일련번호를 통한 조회, 처음부터 10개까지 범위 조회
df.iloc[2]		# 2번째 항목 조회

df.loc['야채':'딸기']		# 이런 식으로 범위 지정도 가능, 야채부터 딸기까지의 범위 조회
# 숫자를 통한 범위 지정은 0부터 시작이므로 끝숫자의 인덱스가 미포함
df.loc['야채':'딸기', '가격']
df.loc['야채':'딸기', ['매출','가격']]		# 출력 범위를 리스트 형식으로도 가능
df.loc['야채':'딸기', '가격':'매출']		# 출력 범위를 범위 지정 형식으로도 가능

df['가격'] > 20000		# 조건을 통한 조회도 가능
mask = df['가격'] > 20000
df[mask]			# 데이터프레임 형식으로 출력하기 위한 응용

mask = (df['가격'] > 5000) & (df['가격'] < 20000)
df[mask]			# 이처럼 다중 조건도 가능

```
