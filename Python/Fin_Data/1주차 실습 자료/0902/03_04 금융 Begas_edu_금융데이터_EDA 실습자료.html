Page 8. Bollinger Band¶

In [26]:
def bb(x, w=20, k=2):
    """
    Calculate Bollinger Bands
    ubb = MA_w(x) + k * sd(x)
    mbb = MA_w(x)
    lbb = MA_w(x) - k * sd(x)
    :param x:
    :return: (ubb, mbb, lbb)
    """
    x = pd.Series(x)
    mbb = x.rolling(w).mean()
    ubb = mbb + k * x.rolling(w).std()
    lbb = mbb - k * x.rolling(w).std()
    ubb.plot(x='Date', y='UBB', color = 'red')
    mbb.plot(x='Date', y='MBB', color = 'black')
    lbb.plot(x='Date', y='LBB', color = 'blue')
    plt.xticks(rotation=45, ha='right')
    plt.show()
